$(function() {
	$("#player1").mediaPlayer({
		debug : true,
		autoplay : true,
		src : "http://www-player-i.ina.fr/medias/demo/audi.mp4",
		callbacks :
			{
				onPlay : 'myOnPlayCallback',
				onPause : 'myOnPauseCallback',
				onComplete : 'myOnCompleteCallback',
				onTimeupdate : 'myOnTimeupdateCallback'
			}

	});
	$("#player1").on(fr.ina.amalia.player.PlayerEventType.TIME_CHANGE, function( e, data ) {
		$('#timeUpdate').html(fr.ina.amalia.player.helpers.UtilitiesHelper.getHours(data.currentTime));
	});
});
function myOnPlayCallback()
{
	$('#myOnPlayCallback').html('received');
}
function myOnPauseCallback()
{
	$('#myOnPauseCallback').html('received');
}
function myOnCompleteCallback()
{
	$('#myOnCompleteCallback').html('received');
}
function myOnTimeupdateCallback(currentTime)
{
	$('#myOnTimeupdateCallback').html('received currentTime: '+currentTime);
}
